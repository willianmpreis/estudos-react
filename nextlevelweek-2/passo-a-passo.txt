1) yarn create react-app web --template typescript ou npx create react-app web --template typescript
2) yarn start
3) yarn add react-router-dom #instalar o router
4) yarn add @types/react-router-dom -D

backend
1) npm init -y
2) npm i typescript -D
3) npx tsc --init, para criar o arquivo tsconfig.json file.
4) npm i ts-node-dev -D, instala uma lib que faz o papel do nodemon

mobile
1) expo init mobile
2) yarn start
3) expo install expo-font @expo-google-fonts/archivo @expo-google-fonts/poppins #pacote para trabalhar com fontes + fontes abertas do google
4 ) expo install react-native-gesture-handler react-native-reanimated react-native-screens react-native-safe-area-context @react-native-community/masked-view #Navegação
5) expo install @react-native-community/async-storage
